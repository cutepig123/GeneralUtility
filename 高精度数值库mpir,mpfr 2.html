3.3. 计算pi 割圓術 (劉徽)

 

#define MY_PRINT(x) printf(#x " "); mpfr_out_str(stdout, 10, 0, x, MPFR_RNDD); putchar('\n');

 

// http://zh.wikipedia.org/zh-hk/%E5%89%B2%E5%9C%86%E6%9C%AF_%28%E5%88%98%E5%BE%BD%29

void TestPi()

{

mpfr_t  n,AB, OP, PC, AC, AP, AP_2, t, t_05, r, pi_m, pi_c;

 

mpfr_inits2(400, n, AB, OP, PC, AC, AP, AP_2, t, t_05, r, pi_m, pi_c, NULL);

mpfr_const_pi(pi_c, 400);

 

mpfr_set_str(n, "6", 10, MPFR_RNDN);

mpfr_set_str(AB, "1", 10, MPFR_RNDN);

mpfr_set_str(r, "1", 10, MPFR_RNDN);

mpfr_set_str(t_05, "0.5", 10, MPFR_RNDN);

MY_PRINT(AB);

MY_PRINT(r);

MY_PRINT(t_05);

 

int i =0;

for (; i < 1000; mpfr_mul_ui(n, n, 2, MPFR_RNDN), i++)

{

MY_PRINT(n);

 

// AP^2

mpfr_mul(AP, AB, t_05, MPFR_RNDN);

//MY_PRINT(AB);

//MY_PRINT(AP);

mpfr_mul(AP_2, AP, AP, MPFR_RNDN);

//MY_PRINT(AP_2);

 

// OP

mpfr_sub(OP, r, AP_2, MPFR_RNDN);

mpfr_sqrt(OP, OP, MPFR_RNDN);

//MY_PRINT(OP);

 

//PC

mpfr_sub(PC, r, OP, MPFR_RNDN);

//MY_PRINT(PC);

 

//AC

mpfr_mul(PC, PC, PC, MPFR_RNDN);

mpfr_add(AC, AP_2, PC, MPFR_RNDN);

mpfr_sqrt(AC, AC, MPFR_RNDN);

//MY_PRINT(AC);

 

// pi

mpfr_mul(pi_m, AC, n, MPFR_RNDN);

MY_PRINT(pi_m);

 

// update AB

mpfr_set(AB, AC, MPFR_RNDN);

}

 

MY_PRINT(pi_c);

;

mpfr_clears(n, AB, OP, PC, AC, AP, AP_2, t, t_05, r, pi_m, pi_c, NULL);

}
